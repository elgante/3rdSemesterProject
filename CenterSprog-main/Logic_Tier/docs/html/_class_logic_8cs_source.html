137 foreach (var participant in participants)

142 LastName = participant.LastName,
143 Email = participant.Email
144 };
145
146 var totalLessons = lessons.Count;
147 var attendedLessons = lessonAttendance
148 .Count(lesson => lesson.Attendees.Any(attendance => attendance.Username == participant.Username));
149
150 var lessonsMissed = totalLessons - attendedLessons;
151 var totalAbsence = totalLessons == 0 ? 0 : (double)lessonsMissed / totalLessons * 100;
152
153 userAttendanceDto.TotalAbsence = totalAbsence;
154
155 participantsWithAttendance.Add(userAttendanceDto);
156 }
