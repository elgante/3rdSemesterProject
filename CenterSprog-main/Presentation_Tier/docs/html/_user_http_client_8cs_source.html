31 private static IEnumerable<Claim> ParseClaimsFromJwt(string jwt)
32 {
33 string payload = jwt.Split('.')[1];
34 byte[] jsonBytes = ParseBase64WithoutPadding(payload);
35 Dictionary<string, object>? keyValuePairs = JsonSerializer.Deserialize<Dictionary<string, object>>(jsonBytes);
36 return keyValuePairs!.Select(kvp => new Claim(kvp.Key, kvp.Value.ToString()!));
37 }
